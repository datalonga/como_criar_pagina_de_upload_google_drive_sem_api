Implementar Upload Google Drive

Sistema completo para arquivos grandes (+1GB), com barra de progresso, sem Google Cloud Console e sem login do usuario.

O que voce vai criar

Um sistema de upload professional com:

  • Upload de arquivos ilimitado (Resumable Upload API)
  • Barra de progresso em tempo real
  • Interface Drag & Drop moderna
  • Zero configuracao de Cloud Console
  • Usuario final sem necessidade de login

Backend (Google Apps Script)

  1. 1
    Acesse script.google.com

    E crie um novo projeto.

  2. 2
    Cole o codigo abaixo

    Substitua o conteudo do arquivo Code.gs.

  3. 3
    Altere o FOLDER_ID

    Coloque o ID da pasta do seu Google Drive onde os arquivos serao salvos.

const FOLDER_ID = 'SEU_FOLDER_ID_AQUI'; 

// Serve a pagina HTML principal
function doGet() {
  return HtmlService.createHtmlOutputFromFile('index')
    .setTitle('Upload Drive')
    .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL)
    .addMetaTag('viewport', 'width=device-width, initial-scale=1');
}

// Retorna o token OAuth2 do script (permissao do proprietario)
function getAccessToken() {
  return ScriptApp.getOAuthToken();
}

// Retorna o ID da pasta configurada
function getFolderId() {
  return FOLDER_ID;
}

// Verifica se a pasta existe (util para debug)
function verifyFolder() {
  try {
    const folder = DriveApp.getFolderById(FOLDER_ID);
    return {
      success: true,
      name: folder.getName(),
      id: folder.getId()
    };
  } catch (e) {
    return {
      success: false,
      error: e.message
    };
  }
}

  <script>
    // Estado da aplicacao
    let selectedFiles = [];
    let accessToken = null;
    let folderId = null;
    let isUploading = false;
    let completedFiles = [];

    // Elementos DOM
    const dropZone = document.getElementById('dropZone');
    const fileInput = document.getElementById('fileInput');
    const fileList = document.getElementById('fileList');
    const progressSection = document.getElementById('progressSection');
    const progressBar = document.getElementById('progressBar');
    const progressPercent = document.getElementById('progressPercent');
    const progressLabel = document.getElementById('progressLabel');
    const progressStatus = document.getElementById('progressStatus');
    const uploadBtn = document.getElementById('uploadBtn');
    const btnText = document.getElementById('btnText');
    const statusMessage = document.getElementById('statusMessage');
    const statusIcon = document.getElementById('statusIcon');
    const statusText = document.getElementById('statusText');
    const completedList = document.getElementById('completedList');
    const completedItems = document.getElementById('completedItems');

    // Inicializacao
    async function init() {
      try {
        // Obter token de acesso e folder ID
        accessToken = await google.script.run.withSuccessHandler(onTokenReceived).getAccessToken();
      } catch (e) {
        showStatus('Erro ao inicializar: ' + e.message, 'error');
      }
    }

    function onTokenReceived(token) {
      accessToken = token;
      google.script.run.withSuccessHandler(onFolderIdReceived).getFolderId();
    }

    function onFolderIdReceived(id) {
      folderId = id;
      console.log('Inicializado com sucesso. Folder ID:', folderId);
    }

    // Formatar tamanho de arquivo
    function formatSize(bytes) {
      if (bytes === 0) return '0 Bytes';
      const k = 1024;
      const sizes = ['Bytes', 'KB', 'MB', 'GB', 'TB'];
      const i = Math.floor(Math.log(bytes) / Math.log(k));
      return parseFloat((bytes / Math.pow(k, i)).toFixed(2)) + ' ' + sizes[i];
    }

    // Event Listeners - Drop Zone
    dropZone.addEventListener('click', () => fileInput.click());

    dropZone.addEventListener('dragover', (e) => {
      e.preventDefault();
      dropZone.classList.add('drag-over');
    });

    dropZone.addEventListener('dragleave', () => {
      dropZone.classList.remove('drag-over');
    });

    dropZone.addEventListener('drop', (e) => {
      e.preventDefault();
      dropZone.classList.remove('drag-over');
      handleFiles(e.dataTransfer.files);
    });

    fileInput.addEventListener('change', (e) => {
      handleFiles(e.target.files);
    });

    // Processar arquivos selecionados
    function handleFiles(files) {
      for (const file of files) {
        if (!selectedFiles.some(f => f.name === file.name && f.size === file.size)) {
          selectedFiles.push(file);
        }
      }
      renderFileList();
      updateUploadButton();
    }

    // Renderizar lista de arquivos
    function renderFileList() {
      fileList.innerHTML = '';
      
      selectedFiles.forEach((file, index) => {
        const item = document.createElement('div');
        item.className = 'file-item';
        item.style.animationDelay = `${index * 0.05}s`;
        item.innerHTML = `
          <div class="file-icon">
            <svg xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 24 24" stroke="currentColor" stroke-width="2">
              <path stroke-linecap="round" stroke-linejoin="round" d="M9 12h6m-6 4h6m2 5H7a2 2 0 01-2-2V5a2 2 0 012-2h5.586a1 1 0 01.707.293l5.414 5.414a1 1 0 01.293.707V19a2 2 0 01-2 2z" />
            </svg>
          </div>
          <div class="file-info">
            <p class="file-name">${file.name}</p>
            <p class="file-size">${formatSize(file.size)}</p>
          </div>
          <button class="file-remove" onclick="removeFile(${index})" aria-label="Remover arquivo">
            <svg xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 24 24" stroke="currentColor" stroke-width="2">
              <path stroke-linecap="round" stroke-linejoin="round" d="M6 18L18 6M6 6l12 12" />
            </svg>
          </button>
        `;
        fileList.appendChild(item);
      });
    }

    // Remover arquivo da lista
    function removeFile(index) {
      selectedFiles.splice(index, 1);
      renderFileList();
      updateUploadButton();
    }

    // Atualizar botao de upload
    function updateUploadButton() {
      if (selectedFiles.length === 0) {
        uploadBtn.disabled = true;
        btnText.textContent = 'Selecione arquivos para upload';
      } else {
        uploadBtn.disabled = isUploading;
        const totalSize = selectedFiles.reduce((acc, f) => acc + f.size, 0);
        btnText.textContent = `Enviar ${selectedFiles.length} arquivo${selectedFiles.length > 1 ? 's' : ''} (${formatSize(totalSize)})`;
      }
    }

    // Upload Button Click
    uploadBtn.addEventListener('click', startUpload);

    // Iniciar upload
    async function startUpload() {
      if (selectedFiles.length === 0 || isUploading) return;

      isUploading = true;
      uploadBtn.disabled = true;
      statusMessage.className = 'status-message';
      completedList.classList.remove('active');
      
      progressSection.classList.add('active');

      for (let i = 0; i < selectedFiles.length; i++) {
        const file = selectedFiles[i];
        progressLabel.textContent = `Enviando ${file.name}...`;
        
        try {
          await uploadFile(file, i, selectedFiles.length);
          completedFiles.push(file.name);
        } catch (error) {
          showStatus(`Erro ao enviar ${file.name}: ${error.message}`, 'error');
          isUploading = false;
          updateUploadButton();
          return;
        }
      }

      // Upload concluido
      isUploading = false;
      showStatus('Todos os arquivos foram enviados com sucesso!', 'success');
      showCompletedFiles();
      
      selectedFiles = [];
      fileList.innerHTML = '';
      updateUploadButton();
      
      progressSection.classList.remove('active');
    }

    // Upload de arquivo individual (Resumable Upload)
    async function uploadFile(file, index, total) {
      return new Promise((resolve, reject) => {
        // Iniciar sessao de upload resumable
        const initiateUrl = 'https://www.googleapis.com/upload/drive/v3/files?uploadType=resumable';
        
        const metadata = {
          name: file.name,
          parents: [folderId]
        };

        fetch(initiateUrl, {
          method: 'POST',
          headers: {
            'Authorization': 'Bearer ' + accessToken,
            'Content-Type': 'application/json'
          },
          body: JSON.stringify(metadata)
        })
        .then(response => {
          if (!response.ok) {
            throw new Error('Falha ao iniciar upload');
          }
          const uploadUrl = response.headers.get('Location');
          if (!uploadUrl) {
            throw new Error('URL de upload nao recebida');
          }
          return uploadToUrl(uploadUrl, file);
        })
        .then(() => {
          resolve();
        })
        .catch(error => {
          reject(error);
        });
      });
    }

    // Upload para URL resumable
    function uploadToUrl(uploadUrl, file) {
      return new Promise((resolve, reject) => {
        const CHUNK_SIZE = 8 * 1024 * 1024; // 8MB por chunk
        let uploadedBytes = 0;
        
        progressStatus.textContent = `Preparando upload...`;

        function uploadChunk() {
          const start = uploadedBytes;
          const end = Math.min(start + CHUNK_SIZE, file.size);
          const chunk = file.slice(start, end);

          fetch(uploadUrl, {
            method: 'PUT',
            headers: {
              'Content-Range': `bytes ${start}-${end - 1}/${file.size}`
            },
            body: chunk
          })
          .then(response => {
            if (response.status === 200 || response.status === 201) {
              // Upload completo
              uploadedBytes = file.size;
              updateProgress(uploadedBytes, file.size);
              resolve();
            } else if (response.status === 308) {
              // Chunk recebido, continuar
              uploadedBytes = end;
              updateProgress(uploadedBytes, file.size);
              progressStatus.textContent = `${formatSize(uploadedBytes)} de ${formatSize(file.size)} enviados`;
              uploadChunk();
            } else {
              throw new Error(`Erro no upload: ${response.status}`);
            }
          })
          .catch(error => {
            // Tentar recuperar posicao
            queryUploadStatus(uploadUrl, file)
              .then(resolve)
              .catch(() => reject(error));
          });
        }

        uploadChunk();
      });
    }

    // Consultar status do upload para retomar
    function queryUploadStatus(uploadUrl, file) {
      return new Promise((resolve, reject) => {
        fetch(uploadUrl, {
          method: 'PUT',
          headers: {
            'Content-Range': `bytes */${file.size}`
          }
        })
        .then(response => {
          if (response.status === 308) {
            const range = response.headers.get('Range');
            if (range) {
              const match = range.match(/bytes=(\d+)-/);
              if (match) {
                const uploadedBytes = parseInt(match[1]) + 1;
                // Retomar upload
                resumeUpload(uploadUrl, file, uploadedBytes, resolve, reject);
                return;
              }
            }
          }
          reject(new Error('Nao foi possivel retomar upload'));
        })
        .catch(reject);
      });
    }

    // Retomar upload
    function resumeUpload(uploadUrl, file, startByte, resolve, reject) {
      const CHUNK_SIZE = 8 * 1024 * 1024;
      let uploadedBytes = startByte;

      function uploadNextChunk() {
        const start = uploadedBytes;
        const end = Math.min(start + CHUNK_SIZE, file.size);
        const chunk = file.slice(start, end);

        fetch(uploadUrl, {
          method: 'PUT',
          headers: {
            'Content-Range': `bytes ${start}-${end - 1}/${file.size}`
          },
          body: chunk
        })
        .then(response => {
          if (response.status === 200 || response.status === 201) {
            uploadedBytes = file.size;
            updateProgress(uploadedBytes, file.size);
            resolve();
          } else if (response.status === 308) {
            uploadedBytes = end;
            updateProgress(uploadedBytes, file.size);
            progressStatus.textContent = `${formatSize(uploadedBytes)} de ${formatSize(file.size)} enviados`;
            uploadNextChunk();
          } else {
            reject(new Error(`Erro no upload: ${response.status}`));
          }
        })
        .catch(reject);
      }

      uploadNextChunk();
    }

    // Atualizar barra de progresso
    function updateProgress(uploaded, total) {
      const percent = Math.round((uploaded / total) * 100);
      progressBar.style.width = percent + '%';
      progressPercent.textContent = percent + '%';
    }

    // Mostrar status
    function showStatus(message, type) {
      statusMessage.className = `status-message ${type}`;
      statusText.textContent = message;

      }
    }
